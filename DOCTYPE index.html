<script>

    const totalDays = 365;

    const challengeGrid = document.getElementById('challenge-grid');

    const modalOverlay = document.getElementById('modal-overlay');

    const modalTitle = document.getElementById('modal-title');

    const modalAmountDisplay = document.getElementById('modal-amount');

   

    let paidDays = new Set();

    let currentSelectingDay = null;

 

    const totalGoal = (totalDays / 2) * (2 + (totalDays * 2));

 

    function init() {

        const saved = localStorage.getItem('365_challenge_icons_progress');

        if (saved) {

            const arr = JSON.parse(saved);

            paidDays = new Set(arr);

        }

 

        renderGrid();

        updateStats();

        // 初始化 Lucide 圖示

        lucide.createIcons();

    }

 

    function renderGrid() {

        challengeGrid.innerHTML = '';

        for (let i = 1; i <= totalDays; i++) {

            const box = document.createElement('div');

            box.className = 'day-box' + (paidDays.has(i) ? ' paid' : '');

           

            const dayNum = document.createElement('span');

            dayNum.className = 'day-num';

            dayNum.innerText = `Day ${i}`;

 

            const amount = document.createElement('span');

            amount.className = 'amount';

            amount.innerText = `$${i * 2}`;

 

            box.appendChild(dayNum);

            box.appendChild(amount);

           

            box.onclick = () => openModal(i);

            challengeGrid.appendChild(box);

        }

    }

 

    function openModal(day) {

        if (paidDays.has(day)) return;

        currentSelectingDay = day;

        modalTitle.innerText = `Day ${day} 支付確認`;

        modalAmountDisplay.innerText = `$${day * 2}`;

        modalOverlay.style.display = 'flex';

        // 彈窗開啟時重新刷一次圖示（確保動態內容中的圖示顯示）

        lucide.createIcons();

    }

 

    function closeModal() {

        modalOverlay.style.display = 'none';

        currentSelectingDay = null;

    }

 

    function confirmPayment() {

        if (currentSelectingDay) {

            paidDays.add(currentSelectingDay);

            saveProgress();

            renderGrid();

            updateStats();

            closeModal();

        }

    }

 

    function saveProgress() {

        localStorage.setItem('365_challenge_icons_progress', JSON.stringify(Array.from(paidDays)));

    }

 

    function updateStats() {

        const count = paidDays.size;

        const currentSum = Array.from(paidDays).reduce((sum, day) => sum + (day * 2), 0);

        const remaining = totalGoal - currentSum;

        const percentage = ((count / totalDays) * 100).toFixed(1);

 

        document.getElementById('paid-days').innerText = `${count} / ${totalDays}`;

        document.getElementById('paid-amount').innerText = `$${currentSum.toLocaleString()}`;

        document.getElementById('remaining-amount').innerText = `$${remaining.toLocaleString()}`;

        document.getElementById('percent-text').innerText = `${percentage}%`;

        document.getElementById('progress-fill').style.width = `${percentage}%`;

    }

 

    window.onload = init;

</script>
